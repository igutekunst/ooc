78 const char* List_to_str(const void* _self) {
79  struct List* self = (struct List*) _self;
80  // TODO don't leak memory.
81 
82  /* // This next part leaks memory like crazy. It's only for a POC
83  * // Future versions will support reference counting, so this would look like
84  * void* frame = alloc(1);
85  * out = new_rc(pool, String, "[");
86  * ...
87  *
88  * // In this case, each _rc function will retain a pointer to any allocated
89  * // objects in the first argument (frame)
90  *
91  * out = append_rc(frame, new_rc(frame, String, "\""));
92  * // We don't want to free the memory since we want to
93  * // return a string, so we retain (increase reference count)
94  * // before releasing the frame
95  * retain(out);
96  *
97  * // This will decrease the refcount, and free all the intermediate strings
98  * used above.
99  * Alternatively, there could be an append_destructive, that modifies the first
100  * argument. This is how lists work in python, but not strings.
101  * release(frame);
102  * return out;
103  */
104 
105  ListItem* item = self->head;
106  const struct String* out = typed_new(String, "[");
107  while (item != NULL) {
108  out = append(out, new(String, "\""));
109  out = append(out, to_String(item->value));
110  out = append(out, new(String, "\""));
111  if (item->next != NULL) {
112  out = append(out, new(String, ", "));
113  }
114  item = item->next;
115  }
116  out = append(out, new(String, "]"));
117  return str(out);
118  return "[]";
119 }
