473  struct HashMap_iter *self = (struct HashMap_iter *) _self;
474  struct HashMap *hash_map = self->hash_map;
475  assert(hash_map);
476  if (self->position < self->hash_map->len) {
477  while (true) {
478  struct HashItem *item_to_yield;
479  if (self->current_item->present) {
480  item_to_yield = self->current_item;
481  if (self->current_item->next) {
482  self->current_item = self->current_item->next;
483  } else {
484  // if we are at the end of the linked list, jump to the
485  // next index into the hash table
486  self->index++;
487  self->current_item = &self->hash_map->items[self->index];
488  }
489 
490  self->position++;

492  } else {
493  // If there is not value in this item, jump to the
494  // next index into the hash table
495  self->index++;
496  self->current_item = &self->hash_map->items[self->index];
497  // We haven't found anything so "recurse"
